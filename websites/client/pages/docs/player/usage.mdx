### 2. Include the player in your app

```tsx copy filename="index.tsx"
import { components } from "./components";
import { ComponentProps } from "@motionly/components";

export default function Home() {
  const [text, setText] = useState("Your text");
  const [seed, setSeed] = useState("sdf");
  const [color, setColor] = useState("#FFFFFF");

  const modifications: Partial<ComponentProps>[] = [
    {
      id: "m5sk14y",
      text,
    },
    {
      id: "r81ebs",
      src: `https://picsum.photos/seed/${seed}/1080/1080`,
    },
    {
      id: "skz7v8m",
      color,
    },
  ];

  return (
    <main>
      <div style={{ maxWidth: "600px", margin: "auto" }}>
        <input
          type="text"
          value={text}
          onChange={(e) => setText(e.target.value)}
        />
        <input
          type="text"
          value={seed}
          onChange={(e) => setSeed(e.target.value)}
        />
        <input
          type="color"
          value={color}
          onChange={(e) => setColor(e.target.value)}
        />
        <Player
          comps={components}
          duration={40}
          fps={30}
          height={1080}
          width={1080}
          style={{ width: `100%` }}
          controls
          modifications={modifications}
        />
      </div>
    </main>
  );
}
```
