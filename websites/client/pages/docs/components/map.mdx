# &lt;Map&gt;

Animate maps with ease

## Usage

```tsx
import { Map, MapProps } from "@motionly/components";

const map: MapProps = {
  type: "map",
  location: {
    lat: 48.85,
    lng: 2.29,
  },
  zoom: 300,
  marker: {
    color: "#FF0000",
    size: 20,
  },
  fill: "#0000FFFF",
  mapUrl: "",
  stroke: "#FFFFFF",
  strokeWidth: 2,
  animation: {
    from: 200,
    to: 400,
    start: 0,
    end: 100,
  },
};

return <Map {...map} />;
```

## API

### type

= `map`

### location

#### lat

latitude of the zoom location

#### lng

longitude of the zoom location

### zoom

zoom level of the map

### marker

#### color

#### size

size of the marker circle

### fill

fill color of the map

### stroke

stroke color of the map

### strokeWidth

stroke width of the map

### mapUrl

Map url to use for the map. The default Map url is https://raw.githubusercontent.com/deldersveld/topojson/master/world-countries.json,
but you can use any map url you want. You can find a more info about maps here: https://www.react-simple-maps.io/docs/map-files/

### animation

#### from

zoom level to start the animation

#### to

zoom level to end the animation

#### start

start time in seconds

#### end

end time in seconds
